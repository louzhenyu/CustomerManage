说明
标准复选框控件
示例
代码
               //创建第一个复选框
                Ext.create('Jit.form.field.Checkbox', {
                    renderTo: 'dvPlaceholder1'
                    , boxLabel: '终端'
                    , name: 'Store'
                });
                //创建第二个复选框
                var checkbox1 = Ext.create('Jit.form.field.Checkbox', {
                    renderTo: 'dvPlaceholder1'
                    , boxLabel: '经销商'
                    , name: 'Distributor'
                });
                //设置值
                checkbox1.jitSetValue(true);
                //获取值
                Ext.create('Jit.button.Button', {
                    renderTo: 'dvPlaceholder1'
                    , text: '是否选中'
                    , handler: function () {
                        alert(checkbox1.jitGetValue());
                    }
                });
