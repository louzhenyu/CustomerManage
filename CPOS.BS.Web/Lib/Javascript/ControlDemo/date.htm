标准日期控件
示例
代码
                Ext.create('Jit.form.field.Date', {
                    id:'data1'
                    , renderTo: 'dvPlaceholder1'
                    , fieldLabel: '开始日期'
                });

                Ext.create('Jit.button.Button', {
                    renderTo: 'dvShowSelectValues1'
                    , text: '查询'
                    ,handler:function(){
                        alert(Ext.getCmp("data1").jitGetValue());
                    }
                })

                //赋值
                Ext.getCmp("data1").jitSetValue('2012-01-01');
