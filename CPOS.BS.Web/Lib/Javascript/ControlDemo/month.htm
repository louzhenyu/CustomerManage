月份选择控件
示例
代码
               //创建控件
                Ext.create('Jit.form.field.Month', {
                    fieldLabel: "销售月份"
                    , id: 'ctlMonth1'
                    , format: 'Y-m'     //Y-m是默认格式,可以不设置
                    , value: '2012-01'  //也可以设置一个Date值,例如value:new Date()
                    , renderTo: 'dvPlaceholder1'
                    , emptyText: '请选择日期'
                });

                Ext.create('Jit.button.Button', {
                    renderTo: 'dvShowSelectValues1'
                    , text: '显示选中值'
                    , handler: function () {
                        alert(Ext.getCmp("ctlMonth1").jitGetValue());
                    }
                });
说明
为月份选择控件设置值
示例
代码
               //创建控件
                Ext.create('Jit.form.field.Month', {
                    fieldLabel: "销售月份"
                    , id: 'ctlMonth2'
                    , format: 'Y-m'     //Y-m是默认格式,可以不设置
                    , value: '2012-01'  //也可以设置一个Date值,例如value:new Date()
                    , renderTo: 'dvPlaceholder2'
                    , emptyText: '请选择日期'
                });
                //创建输入框
                Ext.create('Jit.form.field.Text', {
                    renderTo: 'dvSetValue'
                    , id: 'ctlText1'
                    , fieldLabel: '指定月份'
                })
                Ext.getCmp("ctlText1").jitSetValue('2013-01');
                //使用文本类型变量设置值
                Ext.create('Jit.button.Button', {
                    renderTo: 'dvSetValue'
                    ,jitSize:'big'
                    , text: '设置为指定值'
                    , handler: function () {
                        Ext.getCmp("ctlMonth2").jitSetValue(Ext.getCmp("ctlText1").jitGetValue());
                    }
                })
                //使用日期类型变量设置值
                Ext.create('Jit.button.Button', {
                    renderTo: 'dvSetValue'
                    , jitSize: 'big'
                    , text: '设置为当前时间'
                    , handler: function () {
                        Ext.getCmp("ctlMonth2").jitSetValue(new Date());
                    }
                })
