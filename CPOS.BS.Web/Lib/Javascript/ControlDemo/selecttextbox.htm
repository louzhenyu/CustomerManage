点选控件
示例
代码
                //创建控件
                var selectBox = Ext.create('Jit.form.field.SelectTextBox', {
                    renderTo: 'samplecontainer'
                       , fieldLabel: '人员选择'
                });
                //监听点选事件
                selectBox.on("onSelect", function () { alert("onSelect! 选择项."); selectBox.jitSetText("张三"); selectBox.jitSetValue("1"); });
