下拉框 - 多选的示例
示例
代码
                    //准备测试数据
                    var store = Ext.create('Ext.data.Store', {
                        fields: ['ChannelID', 'ChannelName'],
                        data: [
                            { ChannelID: '1', ChannelName: '超市' },
                            { ChannelID: '2', ChannelName: '大卖场' },
                            { ChannelID: '3', ChannelName: '夫妻老婆店夫妻老婆店' },
                            { ChannelID: '4', ChannelName: '便利店' },
                            { ChannelID: '5', ChannelName: '现代通路' },
                            { ChannelID: '6', ChannelName: '中型超市' },
                            { ChannelID: '7', ChannelName: '网上商城' }
                        ]
                    });
                    //创建多选下拉框
                    Ext.create('Jit.form.field.ComboBox', {
                        renderTo: 'dvPlaceholder1'
                        , id: 'cmbMultiSelect'
                        , valueField: 'ChannelID'
                        , displayField: 'ChannelName'
                        , store: store
                        , multiSelect: true
                        , fieldLabel:'渠道'
                        , jitAllText :'全选'  //设置全选的文本内容，默认为All
                    });
                    //创建按钮显示选中值
                    Ext.create('Jit.button.Button', {
                        text: '显示选中值'
                        , renderTo: 'dvShowSelectValues1'
                        , handler: function () {
                            var selectedValues = Ext.getCmp('cmbMultiSelect').getValue();
                            alert(selectedValues);
                        }
                    });
